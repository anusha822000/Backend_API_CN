--35 list the employees who draws salary more than the average salary on their department.
SELECT E.ENAME,E.JOB,E.SAL
FROM EMP E
WHERE SAL>(SELECT AVG(SAL)
			FROM EMP E
			WHERE E.DEPTNO=E.DEPTNO);

--36 list the employees who draws salary more than the average salary on their department sort it based on department name

SELECT E.ENAME,E.JOB,E.SAL
FROM EMP E,DEPT D
WHERE SAL>(SELECT AVG(SAL)
			FROM EMP E,DEPT D
			WHERE E.DEPTNO=E.DEPTNO)
			ORDER BY D.DNAME


--37 Write a query that will give you the names and jobs of all employees in New York with a commission above 1000 

SELECT E.ENAME,E.JOB
FROM EMP E,DEPT D
WHERE  D.LOC='NEWYORK' AND E.COMM>1000

--38 How many employees work in Chicago?
SELECT COUNT(*) AS num_employees_in_chicago
FROM EMP E
JOIN DEPT D 
ON E.DEPTNO = D.DEPTNO
WHERE D.LOC = 'Chicago';
